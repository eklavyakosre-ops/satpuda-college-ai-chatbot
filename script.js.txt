// Send message function

function sendMessage(){

let input = document.getElementById("userInput").value;

if(input.trim() === "") return;

let chatbox = document.getElementById("chatbox");


// user message

chatbox.innerHTML += `
<div class="user-message">
${input}
</div>
`;


// bot reply

let reply = getBotReply(input);

setTimeout(()=>{

chatbox.innerHTML += `
<div class="bot-message">
${reply}
</div>
`;

chatbox.scrollTop = chatbox.scrollHeight;

},500);


document.getElementById("userInput").value="";

}



// chatbot logic

function getBotReply(input){

input = input.toLowerCase();


// courses

if(input.includes("course") || input.includes("courses")){

return "Satpuda College offers B.Tech, Diploma, Polytechnic, MBA, M.Tech, BCA, BBA, B.Sc, BA and B.Com courses.";

}


// streams

else if(input.includes("stream") || input.includes("branch")){

return "Engineering streams include Mechanical Engineering, Civil Engineering, Computer Science, Electrical Engineering and Mining Engineering.";

}


// fees

else if(input.includes("fees")){

return "Polytechnic total fees is about ₹75,000. B.Tech total fees is about ₹1.56 Lakhs. First year Computer Science fees is around ₹39,000.";

}


// placement

else if(input.includes("placement")){

return "Companies visit the college in 7th and 8th semester. Maximum package is around ₹8 LPA. Students also get internship opportunities.";

}


// faculty

else if(input.includes("faculty") || input.includes("teacher")){

return "Faculty student ratio is about 1:20. Teachers are supportive. Dr. Amit Shukla is the HOD of Computer Science.";

}


// timing

else if(input.includes("timing") || input.includes("time")){

return "College timing is 10 AM to 5 PM.";

}


// scholarship

else if(input.includes("scholarship")){

return "OBC scholarship is around ₹25,000 to ₹28,000. SC/ST scholarship is about ₹45,000 to ₹50,000.";

}


// internship

else if(input.includes("internship")){

return "Students get internship opportunities through platforms like Internshala and other companies.";

}


// default

else{

return "Sorry, I didn't understand. Please ask about courses, fees, placement, faculty or timing.";

}

}
